
//初始化远程仓库
$ git init      
 
//添加文件到暂存太
$ git add .
 
//提交代码到本地
$ git commit -m "提交文件"
 
//连接远程仓库
$ git remote add origin https://github.com/huyande/message.git
 
//推送代码到远程仓库
$ git push -u origin master
 
#强制覆盖
$ git push -f -u origin master
